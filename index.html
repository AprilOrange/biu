Biu

Basic Usage

biu('text you want to show')
biu('success', 'text with a type')
biu('info', 'do not auto hide me', false)

Advanced

// global settings
window.biuOpts = {
  height: '50px',
  lineHeight: '50px',
  top: '-55px',
  closeButton: 'x'
}
// use object as argument
biu({
  text: 'hello there, done registered!',
  type: 'success',
  autoFade: false,
  delay: 5000,
  parent: 'body'
})
